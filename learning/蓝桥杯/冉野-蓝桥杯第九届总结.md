### 第九题

```c++
#include<iostream>
#include<vector>
#include<string>
using namespace std;

//类宽度优先搜索
int cg(vector<string>&vi){
    int pos[4][2]={{1,0},{-1,0},{0,-1},{0,1}};
    int ans=0;
    bool bo=false;
    for(int i=0;i<vi.size();i++){
        for(int j=0;j<vi[0].size();j++){
            if(vi[i][j]=='#'){
                int x=i,y=j;
                bo=false;   
                for(int k=0;k<4;k++){
                    int newX=x+pos[k][0];
                    int newY=y+pos[k][1];
                    if(newX>=0&&newX<vi.size()&&newY>=0&&newY<vi[i].size())
                    if(vi[newX][newY]=='.'){
                        bo=true;
                        break;
                    }
                }
                if(bo==false)
                    ans++;
            }
        }
    }
    cout<<ans<<endl;
    return ans;
}

int main(){
    vector<string>vs;
    string s;
    int n=0;
    cin>>n;
    while(n--){
        cin>>s;
        vs.push_back(s);
    }
    cg(vs);
    return 0;
}
```
